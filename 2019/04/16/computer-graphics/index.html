int clip(float p, float q, float* t1, float* t2){
    float t;
    int sign = 1;
    if(p < 0){
        t = q / p;
        if(t > *t2)
            sign = 0;
        else if(t > *t1)
            *t1 = t;
    }
    else if (p > 0){
        t = q / p;
        if(t < *t1)
            sign = 0;
        else if(t < *t2)
            *t2 = t;
    }
    else if(q < 0) sign = 0; 
    return sign;
}
